19  public class SwaggerDiff {

21      public static final String SWAGGER_VERSION_V2 = "2.0";
22  
23      private static Logger logger = LoggerFactory.getLogger(SwaggerDiff.class);
24  
25      private Swagger oldSpecSwagger;
26      private Swagger newSpecSwagger;
27  
28      private List<Endpoint> newEndpoints;
29      private List<Endpoint> missingEndpoints;
30      private List<ChangedEndpoint> changedEndpoints;

32      /**
33       * compare two swagger 1.x doc
34       * 
35       * @param oldSpec
36       *            old api-doc location:Json or Http
37       * @param newSpec
38       *            new api-doc location:Json or Http
39       * @return swagger differences
40       */
41      public static SwaggerDiff compareV1(String oldSpec, String newSpec) {
42          return compare(oldSpec, newSpec, null, null);
43      }
44  
45      /**
46       * compare two swagger v2.0 doc
47       * 
48       * @param oldSpec
49       *            old api-doc location:Json or Http
50       * @param newSpec
51       *            new api-doc location:Json or Http
52       * @return swagger differences
53       */
54      public static SwaggerDiff compareV2(String oldSpec, String newSpec) {
55          return compare(oldSpec, newSpec, null, SWAGGER_VERSION_V2);
56      }
57

59      /**
60       * compare two swagger v2.0 Sring
61       *
62       * @param oldSpec old api-doc json as string
63       * @param newSpec new api-doc json as string
64       * @return swagger differences
65       */
66      public static SwaggerDiff compareV2Raw(String oldSpec, String newSpec) {
67          return new SwaggerDiff(oldSpec, newSpec).compare();
68      }
69      
70      /**
71       * Compare two swagger v2.0 docs by JsonNode
72       *
73       * @param oldSpec
74       *            old Swagger specification document in v2.0 format as a JsonNode
75       * @param newSpec
76       *            new Swagger specification document in v2.0 format as a JsonNode
77       * @return swagger differences
78       */
79      public static SwaggerDiff compareV2(JsonNode oldSpec, JsonNode newSpec) {
80          return new SwaggerDiff(oldSpec, newSpec).compare();
81      }
82  
83      public static SwaggerDiff compare(String oldSpec, String newSpec,
84              List<AuthorizationValue> auths, String version) {
85          return new SwaggerDiff(oldSpec, newSpec, auths, version).compare();
86      }
87  
88      private SwaggerDiff(String rawOldSpec, String rawNewSpec) {
89          SwaggerParser swaggerParser = new SwaggerParser();
90          oldSpecSwagger = swaggerParser.parse(rawOldSpec);
91          newSpecSwagger = swaggerParser.parse(rawNewSpec);
92  
93          if (null == oldSpecSwagger || null == newSpecSwagger) {
94              throw new RuntimeException("cannot read api-doc from spec.");
95          }
96      }
97  
98      private SwaggerDiff(String oldSpec, String newSpec, List<AuthorizationValue> auths,
99              String version) {
100         if (SWAGGER_VERSION_V2.equals(version)) {
101             SwaggerParser swaggerParser = new SwaggerParser();
102             oldSpecSwagger = swaggerParser.read(oldSpec, auths, true);
103             newSpecSwagger = swaggerParser.read(newSpec, auths, true);
104         } else {
105             SwaggerCompatConverter swaggerCompatConverter = new SwaggerCompatConverter();
106             try {
107                 oldSpecSwagger = swaggerCompatConverter.read(oldSpec, auths);
108                 newSpecSwagger = swaggerCompatConverter.read(newSpec, auths);
109             } catch (IOException e) {
110                 logger.error("cannot read api-doc from spec[version_v1.x]", e);
111                 return;
112             }
113         }
114         if (null == oldSpecSwagger || null == newSpecSwagger) { throw new RuntimeException(
115                 "cannot read api-doc from spec."); }
116     }
117 
118     private SwaggerDiff(JsonNode oldSpec, JsonNode newSpec) {
119         SwaggerParser swaggerParser = new SwaggerParser();
120         oldSpecSwagger = swaggerParser.read(oldSpec, true);
121         newSpecSwagger = swaggerParser.read(newSpec, true);
122         if (null == oldSpecSwagger || null == newSpecSwagger) { throw new RuntimeException(
123             "cannot read api-doc from spec."); }
124     }
125 
126     private SwaggerDiff compare() {
127     	SpecificationDiff diff = SpecificationDiff.diff(oldSpecSwagger, newSpecSwagger);
128         this.newEndpoints = diff.getNewEndpoints();
129         this.missingEndpoints = diff.getMissingEndpoints();
130         this.changedEndpoints = diff.getChangedEndpoints();
131         return this;
132     }
133 
134     public List<Endpoint> getNewEndpoints() {

137 
138     public List<Endpoint> getMissingEndpoints() {
139         return missingEndpoints;
140     }
141 
142     public List<ChangedEndpoint> getChangedEndpoints() {
143         return changedEndpoints;
144     }

146     public String getOldVersion() {
147         return oldSpecSwagger.getInfo().getVersion();

150     public String getNewVersion() {
151         return newSpecSwagger.getInfo().getVersion();
152     }
