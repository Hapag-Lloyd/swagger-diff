3   import java.lang.reflect.Field;
4   import java.util.regex.Pattern;
5   
6   import com.beust.jcommander.IParameterValidator2;
7   import com.beust.jcommander.ParameterDescription;
8   import com.beust.jcommander.ParameterException;
9   import com.beust.jcommander.Parameterized;
10  
11  public class RegexValidator implements IParameterValidator2 {
12  
13      private static final String PARAMETERIZED_FIELD_NAME = "field";
14  
15      @Override
16      public void validate(String name, String value) throws ParameterException {
17          return;
18      }
19  
20      @Override
21      public void validate(String name, String value, ParameterDescription pd)
22              throws ParameterException {
23          Parameterized parameterized = pd.getParameterized();
24          Class<? extends Parameterized> clazz = parameterized.getClass();
25          try {
26              Field declaredField = clazz.getDeclaredField(PARAMETERIZED_FIELD_NAME);
27              declaredField.setAccessible(true);
28              Field paramField = (Field) declaredField.get(parameterized);
29              Regex regex = paramField.getAnnotation(Regex.class);
30              if (null == regex) return;
31              String regexStr = regex.value();
32              if (!Pattern.matches(regexStr, value)) { throw new ParameterException(
33                      "Parameter " + name + " should match " + regexStr + " (found " + value + ")"); }
34          } catch (NoSuchFieldException e) {
35              return;
36          } catch (IllegalArgumentException e) {
37              return;
38          } catch (IllegalAccessException e) {
39              return;
40          }
41      }
42  
43  }
