12  public class PropertyDiff {
13  
14      private List<ElProperty> increased;
15      private List<ElProperty> missing;
16      private List<ElProperty> changed;
17  
18      Map<String, Model> oldDedinitions;
19      Map<String, Model> newDedinitions;
20  
21      private PropertyDiff() {
22          increased = new ArrayList<ElProperty>();
23          missing = new ArrayList<ElProperty>();
24          changed = new ArrayList<ElProperty>();
25      }
26  
27      public static PropertyDiff buildWithDefinition(Map<String, Model> left, Map<String, Model> right) {
28          PropertyDiff diff = new PropertyDiff();
29          diff.oldDedinitions = left;
30          diff.newDedinitions = right;
31          return diff;
32      }
33  
34      public PropertyDiff diff(Property left, Property right) {
35          ModelDiff diff = ModelDiff.buildWithDefinition(oldDedinitions, newDedinitions).diff(left, right);
36          increased.addAll(diff.getIncreased());
37          missing.addAll(diff.getMissing());
38          changed.addAll(diff.getChanged());
39          return this;
40      }
